dt binding example:


	
	
	card_reader {
		compatible = "card-reader,input";
	
		data-gpios = <&pio 1 4 GPIO_ACTIVE_HIGH>; /* PB4 */
	
		interrupt-parent = <&pio>;
		interrupts = <1 4 IRQ_TYPE_EDGE_FALLING>, <1 5 IRQ_TYPE_EDGE_FALLING>;
		interrupt-names = "data0", "data1";
	};
		
	em125 {
		compatible = "em125-reader,input";
		
		pwms = <&pwm 1 8000 1>;

		interrupt-parent = <&r_pio>;
		interrupts = <0 4 IRQ_TYPE_NONE>;
		interrupt-names = "data";
	};
